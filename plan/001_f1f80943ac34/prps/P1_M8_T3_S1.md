### Wiring helper + gates (integration_real_test.go)
```go
package generate
func requireReal(t *testing.T) {
    t.Helper()
    if os.Getenv("STAGEHAND_RUN_REAL") != "1" {
        t.Skip("skipping real-agent test: set STAGEHAND_RUN_REAL=1 (and -tags integration_real)")
    }
}
func requireInstalled(t *testing.T, name string) provider.Manifest {
    t.Helper()
    reg := provider.NewRegistry(provider.Builtins(), nil)
    if !reg.Detect()[name] {
        t.Skipf("skipping: agent %q not on $PATH", name)
    }
    m, _ := reg.Get(name) 
    return m
}
func realDeps(t *testing.T, dir, name string, cfg config.Config) (Deps, *bytes.Buffer, *bytes.Buffer) {
    m := requireInstalled(t, name)
    g, _ := git.New(dir)
    stdout, stderr := &bytes.Buffer{}, &bytes.Buffer{}
    return Deps{
        Git: g, Runner: provider.NewExecutor(dir), Manifest: m, Config: cfg,
        Output: ui.NewOutput(stdout, stderr, false, true),
    }, stdout, stderr
}

```

### Per-agent e2e test shape (one per agent; pi shown)
```go
func TestIntegrationReal_Pi(t *testing.T) {
    requireReal(t)
    dir := newTempRepo(t)                 
    seedCommit(t, dir, "feat: baseline")  
    writeStage(t, dir, "feature.go", "package x\n\n// new feature\n")
    cfg := config.Default(); cfg.Timeout = 10 * time.Minute
    deps, _, _ := realDeps(t, dir, "pi", cfg)
    res, err := CommitStaged(context.Background(), deps)
    if err != nil { t.Fatalf("CommitStaged(pi): %v", err) }
    if res.CommitSHA == "" { t.Fatal("empty CommitSHA") }
    if commitType(t, dir, res.CommitSHA) != "commit" { t.Error("not a commit object") }
    if strings.TrimSpace(res.Message) == "" { t.Error("empty message") }
    if res.Subject == "feat: baseline" { t.Error("subject duplicates seeded commit") }
    if headSHA(t, dir) != res.CommitSHA { t.Error("HEAD did not advance to the commit") }
}

```
